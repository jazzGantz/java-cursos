import java.util.Scanner;// el progrma utiliza la clase Scanner
public class comparacion{
   //el metodo main empieza la ejecucion de la aplicacion
  public static void main(String[] args){
   //crea objeto Scanner para obtener la entrada de la ventana de comandos
   Scanner sc = new Scanner(System.in);
   
   int numero1 //1er numero acomparar
   int numero2 //2do numero a comparar
    
   System.out.println("Escribe el primer entero :"); //indicador
   numero1 = sc.nextInt();
   System.out.println("Escribe el segundo numero"); //indicador
   numero2 = sc.nextInt(); //lee el segundo numero de usuario.
   if(numero1 ==numero2){
      System.out.printf("%d%","%d%n", numero1, numero2);
    }
    
   if(numero1 != numero2){
     System.out.printf("%d%","%d%n", numero1, numero2);
    }
  
   if(numero1 < numero2){
     System.out.printf("%d%","%d%n", numero1, numero2);
   }
  
   if(numero1 > numero2){
     System.out.printf("%d%","%d%n", numero1, numero2);
   }  
   
   if(numero1 <= numero2){
     System.out.printf("%d%","%d%n", numero1, numero2);
   }  
    
   if(numero1 >= numero2){
     System.out.printf("%d%","%d%n", numero1, numero2);
   }    
 }//fin metodo main
}// fin metodo clase Comparativa
