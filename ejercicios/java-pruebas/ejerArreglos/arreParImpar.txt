import java.util.Scanner;
 //obtener cuantos numeros son pares y imparesen un arreglo

public class arreParImpar{
 public static void main(String[] args){
  int arrays[] = new int[10];

  int incremento = 1; 
  for(int i = 0; i <arrays.length; i++){
   arrays[i] = incremento;
    incremento++;
  }

  for(j =0; j < arrays.length; j++){
     if( arrays[j] % 2 == 0){
        
        System.out.println("numeros pares :"+arrays[j]);
     }else{
      System.out.println("numeros impares :"arrays[j]);
     }
   }
 }
}
